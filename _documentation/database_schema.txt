Table rank as C {
  id int [pk, increment]
  name varchar
  slug slug
  parent varchar
  is_active boolean
}

Table IPCRType as PT {
  id int [pk, increment]
  name varchar
  is_active boolean
}

Table IPCRSpecification as PS {
  id int [pk, increment]
  ipcr_type int
  name varchar
} 
Ref: PT.id < PS.ipcr_type  

Table IPCR as P {
  id int [pk, increment]
  ipcr_type int
  rank int
  description varchar
  slug slug
  regular_price decimal
  discount_price decimal
  is_active boolean
  created_at timestamp
  updated_at timestamp
}
Ref: PT.id < P.ipcr_type 
Ref: C.id < P.rank

Table IPCRSpecificationValue as PSV {
  id int [pk, increment]
  ipcr int
  specification int
  value varchar
}
Ref: P.id < PSV.ipcr
Ref: PS.id < PSV.specification

}
